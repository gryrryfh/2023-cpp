## 전위 ++ 연산자 오버로딩, 후위 ++ 연산자 오버로딩의 예를 만들고 설명을 적으시오.
``` c++
class Counter {
private:
    int count;

public:
    Counter() : count(0) {}

    // 전위 증가 연산자
    Counter operator++() {
        ++count;
        return *this;
    }

    // 후위 증가 연산자
    Counter operator++(int) {
        Counter temp = *this;
        ++count;
        return temp;
    }

    int getCount() const {
        return count;
    }
};
```
