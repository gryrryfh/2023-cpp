## CPoly로 부터 w, h를 상속받고,사각형의 면적을 구하는 코딩을 하시오. 포인터를 활용하여 화면에 출력하시오.
#include <iostream>
using namespace std;
class CPoly{
	protected:
		int w,h;
		public:};
			
class CRect : public CPoly{
		public:
		CRect(){w=2;h=3;}
		void Area(){cout << w*h;}
		};
		int main(){
			CRect r;
			CRect*p =&r;
			p->Area(); //포인터 연산자//
			return 0;
		}
		//포인터의 값은 *변수//
		//포인터는 원하는 메모리지점을 가르킨다//
		//포인터 퀴즈 3.29//
