## CPoly로 부터 w, h를 상속받고,사각형의 면적을 구하는 코딩, 포인터를 활용하여 화면에 출력
```
#include <iostream>
using namespace std;
class CPoly{
	protected:
		int w,h;
		public:};
			
class CRect : public CPoly{
		public:
		CRect(){w=2;h=3;}
		void Area(){cout << w*h;}
		};
		int main(){
			CRect r;
			CRect*p =&r;
			p->Area(); //포인터 연산자//
			return 0;
		}
		//포인터의 값은 *변수//
		//포인터는 원하는 메모리지점을 가르킨다//
```

## 클래스에서 static을 활용하는 예
```
#include <iostream>
using namespace std;
void count(){
	static int  cnt=0; //static은 1번만 초기화한다, 메모리변수에 자리를 잡는다, 로컬변수처럼 사용//
	cnt++;
	cout << cnt << endl;
}
int main(){
	count();
	count();
	count();
	return 0;
}
```

## friend를 활용하는 클래스의 예
```
#include <iostream>
using namespace std;
class CTest{
	int x;
	private:
		CTest(int a=2):x(a){}
		friend int main();
};
int main(){
	CTest me;
	cout << me.x << endl;
	return 0;
}
```
