## CPnt는 두 지점을 나타내는 클래스 이다. 	CPnt p1, p2, p3; 에서 p1의 지점과 p2의 지점이 더해 지는 연산에서 각각의 h와 w가 더해 질 수 있도록 오퍼레이터 오버로딩을 하시오.
```
#include<iostream>
using namespace std;
class CPnt{
	private: int w,h;
	public:
		CPnt(int a, int b){
			w = a , h = b;
		}
		CPnt(){
			w=0,h=0;
		}
		void Pr(){
			cout << w << "," << h << endl;
		}
		void Prt(){
			cout << w*h << endl;
		}
		CPnt Add(CPnt t){			// 오퍼레이터 오버로딩 사용전
			CPnt r;
			r.w = w + t.w;
			r.h = h + t.h;
			return r;
		}
};
int main(){
	CPnt p1(2,2), p2(3,3), p3;
	p3 = p1.Add(p2);				// 오퍼레이터 오버로딩 사용전
	p3.Pr();
	p3.Prt();
	return 0;
}
```

	void pr(){
			cout << w << ',' << h << endl;
	}
	CPnt operator+(CPnt r){
		CPnt p3;
		p3.w = w +r.w;
		p3.h = h+ r.h;
		return (p3);
	}
};

int main(){
	CPnt p1(1,1), p2(2,2),  p3;
	p3=p1.operator+(p2);
	p1.pr();
	p2.pr();
	p3.pr();
	return 0;
}
