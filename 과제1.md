using namespace std;
class CRect{int w, h;
public:
	CRect(int _w, int _h){
	w= _w; h= _h;}
	CRect(){w=5, h=6;}
	void pr(){
		cout << w * h << endl;
	
	}
	void prpt(){
			cout << w << ',' << h << endl;
	}
	CRect operator+(CRect r){
		CRect t;
		t.w = w +r.w;
		t.h = h+ r.h;
		return (t);
	}
};

int main(){
	CRect a(1,1), b(2,2), c, t;

	a.pr();
	b.pr();
	c.pr();
	t.pr();

	b.prpt();
